Public Function ErrSyori(DataErr As Integer) As Integer
    Select Case DataErr
        Case 2169   'フォーム終了時の更新失敗　MsgBoxによる処理不可
            ErrSyori = acDataErrDisplay
        Case 2237   'コンボのデーターチェックエラー
            MsgBox "入力された値は正しくありません" & "(" & DataErr & ")", , "入力失敗"
            ErrSyori = acDataErrContinue
        Case 2473   'フォーム終了時の失敗　MsgBoxによる処理不可
            Call SysCmd(acSysCmdSetStatus, "フォーム終了時の失敗")
            ErrSyori = acDataErrContinue
        Case 3022
            MsgBox "同じキーを持つレコードが有ります" & "(" & DataErr & ")", , "レコード更新失敗"
            ErrSyori = acDataErrContinue
        Case 3058
            Call SysCmd(acSysCmdSetStatus, "レコード保存失敗")
            ErrSyori = acDataErrContinue
        Case 3314   'Nullを設定した
            MsgBox "入力を省略できません" & "(" & DataErr & ")", , "入力失敗"
            ErrSyori = acDataErrContinue
        Case 8515   'クリップボードのデータが多すぎる　MsgBoxによる処理不可
            ErrSyori = acDataErrDisplay
        Case Else
            Call SysCmd(acSysCmdSetStatus, Error(DataErr) & "(" & DataErr & ")")
            ErrSyori = acDataErrDisplay
    End Select
End Function

Private Function IsOpen(strForm As String)
    IsOpen = (SysCmd(acSysCmdGetObjectState, acForm, strForm) > 0)
End Function

Function ahtGetDate(varDate As Variant) As Variant

    Const ahtcCalForm = "frmPopupCal"

    ' Open calendar form in dialog mode
    ' passing it the current date using OpenArgs
    DoCmd.OpenForm ahtcCalForm, WindowMode:=acDialog, OpenArgs:=Nz(varDate)

    ' Check if the form is open; if so return the date
    ' selected in the calendar control and close the
    ' popup calendar form and pass the new date back
    ' to the control. Otherwise, just return a null.
    If IsOpen(ahtcCalForm) Then
        ahtGetDate = Forms(ahtcCalForm).CalDate
        DoCmd.Close acForm, ahtcCalForm
    Else
        ahtGetDate = Null
    End If

End Function

Public Function gf計算値() As Currency
    If gcKeisan(0) = 0 Then
        gf計算値 = 0
    Else
        gf計算値 = Format(gcKeisan(Form_設計Ｂメンテ.設計明細NO) / gcKeisan(0) * 10000) / 100
    End If
End Function
