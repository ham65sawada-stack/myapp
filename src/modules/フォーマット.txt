Public Function FormatStr(avString As Variant, aiLength) As String
    If IsNull(avString) Then
        FormatStr = Space(aiLength)
    Else
        FormatStr = StrConv(MidB(StrConv(avString, vbFromUnicode), 1, aiLength), vbUnicode)
    End If
End Function

Public Function FormatVal(acValue As Currency, aiLength, Optional avZero As String) As String
    If IsMissing(avZero) Then
        FormatVal = Right$(Space(aiLength) + Format(acValue), aiLength)
    ElseIf avZero = "" Then
        FormatVal = Right$(Space(aiLength) + Format(acValue), aiLength)
    Else
        FormatVal = Right$(String(aiLength, avZero) + LTrim$(Format(acValue)), aiLength)
    End If
End Function

Private Function MakeInsertSQL(asTargetTable As String, asSourceTable As String) As String
    Dim wspSpc      As DAO.Workspace
    Dim dbDB        As DAO.Database
    Dim daoTblDef   As DAO.TableDef
    Dim daoQryDef   As DAO.QueryDef
    Dim daoFLD      As DAO.Field

    Set wspSpc = DBEngine.Workspaces(0)
    Set dbDB = CurrentDb

    On Error GoTo ErrMakeIns

    Set daoTblDef = dbDB.TableDefs(asTargetTable)
    With daoTblDef
        MakeInsertSQL = "INSERT INTO " & asTargetTable & "("

        For Each daoFLD In .Fields
            With daoFLD
                MakeInsertSQL = MakeInsertSQL & .Name
            End With
            MakeInsertSQL = MakeInsertSQL & ","
        Next daoFLD
        Mid$(MakeInsertSQL, Len(MakeInsertSQL), 1) = ") "
    End With
    Set daoTblDef = Nothing

    On Error Resume Next

    Set daoTblDef = dbDB.TableDefs(asSourceTable)
    If Err.Number > 0 Then
        Set daoQryDef = dbDB.QueryDefs(asSourceTable)
        With daoQryDef
            MakeInsertSQL = MakeInsertSQL & " SELECT "

            For Each daoFLD In .Fields
                With daoFLD
                    MakeInsertSQL = MakeInsertSQL & .Name
                End With
                MakeInsertSQL = MakeInsertSQL & ","
            Next daoFLD
            Mid$(MakeInsertSQL, Len(MakeInsertSQL), 1) = " "

            .Close
        End With
        Set daoTblDef = Nothing
    Else
        With daoTblDef
            MakeInsertSQL = MakeInsertSQL & " SELECT "

            For Each daoFLD In .Fields
                With daoFLD
                    MakeInsertSQL = MakeInsertSQL & .Name
                End With
                MakeInsertSQL = MakeInsertSQL & ","
            Next daoFLD
            Mid$(MakeInsertSQL, Len(MakeInsertSQL), 1) = " "
        End With
        Set daoTblDef = Nothing
    End If

    MakeInsertSQL = MakeInsertSQL & " FROM [" & asSourceTable & "]"

    Exit Function
ErrMakeIns:
    Set daoQryDef = Nothing
End Function

Public Sub 試編出荷データワークの属性チェック(ByRef aoForm As Form)
    Dim i As Integer
    With aoForm
        If Not IsNumeric(!連番) Then !連番 = 0
        If Not IsDate(!伝票日付) Then !伝票日付 = Date
        If Not IsNumeric(!設計NO) Then !設計NO = 0
        If IsNull(!品番) Then !品番 = ""
        If Not IsNumeric(!メーカ) Then !メーカ = 0
        If IsNull(!気付) Then !気付 = ""
        If IsNull(!担当者名) Then !担当者名 = ""
        If IsNull(!写) Then !写 = ""
        If IsNull(!出荷先) Then !出荷先 = ""
        If IsNull(!出荷先名) Then !出荷先名 = ""
        If IsNull(!出荷元) Then !出荷元 = ""
        For i = 1 To 10
            If IsNull(aoForm("品番" & i)) Then aoForm("品番" & i) = ""
            If IsNull(aoForm("反番" & i)) Then aoForm("反番" & i) = ""
            If Not IsNumeric(aoForm("巾" & i)) Then aoForm("巾" & i) = 0
            If Not IsNumeric(aoForm("長" & i)) Then aoForm("長" & i) = 0
            If Not IsNumeric(aoForm("重量" & i)) Then aoForm("重量" & i) = 0
            If Not IsNumeric(aoForm("目付" & i)) Then aoForm("目付" & i) = 0
            If Not IsNumeric(aoForm("落傷" & i)) Then aoForm("落傷" & i) = 0
            If Not IsNumeric(aoForm("針折" & i)) Then aoForm("針折" & i) = 0
            If Not IsNumeric(aoForm("ハネ" & i)) Then aoForm("ハネ" & i) = 0
            If Not IsNumeric(aoForm("その他" & i)) Then aoForm("その他" & i) = 0
            If Not IsNumeric(aoForm("カット" & i)) Then aoForm("カット" & i) = 0
            aoForm("罰計" & i) = aoForm("落傷" & i) + aoForm("針折" & i) + aoForm("ハネ" & i) + aoForm("その他" & i)
            If IsNull(aoForm("明細備考" & i)) Then aoForm("明細備考" & i) = ""
        Next i
        For i = 1 To 6
            If IsNull(aoForm("原糸備考" & i)) Then aoForm("原糸備考" & i) = ""
            If IsNull(aoForm("備考" & i)) Then aoForm("備考" & i) = ""
        Next i
        If Not IsNumeric(!単価) Then !単価 = 0
    End With
End Sub
