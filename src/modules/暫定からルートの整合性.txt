Public Sub 暫定からルート更新(db As DAO.Database, lng暫定設計ID As Long)
    Dim str生機品番 As String
    Dim str社内備考メモ As String
    str生機品番 = ""
    str社内備考メモ = ""
    Dim rs As DAO.Recordset
    Set rs = db.OpenRecordset("select * from 設計Ｈ社内備考 where 暫定設計ID=" & lng暫定設計ID & ";")
    Dim rsH As DAO.Recordset
    Set rsH = db.OpenRecordset("select * from 設計Ｈ where 設計ID=" & lng暫定設計ID & ";")
    Dim rsH社内備考 As DAO.Recordset
    Set rsH社内備考 = db.OpenRecordset("select * from 設計Ｈ社内備考 where 設計ID=" & lng暫定設計ID & ";")
    Dim cnt As Long
    cnt = 0
    Do Until rs.EOF
        cnt = cnt + 1
        If cnt = 1 Then
            str生機品番 = Trim(IfString(rsH!生機品番))
            str社内備考メモ = Trim(IfString(rsH社内備考!社内備考メモ))
        End If
        Dim lngRoot設計ID As Long
        lngRoot設計ID = val(IfNumeric(rs!設計ID))
        Call 設計H社内備考更新(db, rsH社内備考, lngRoot設計ID)

        Dim cur生産量反月 As Currency
        Dim cur生産量K月 As Currency
        cur生産量反月 = 0
        cur生産量K月 = 0
        Call 契約B更新と生産量計算(db, lng暫定設計ID, lngRoot設計ID, rsH, cur生産量反月, cur生産量K月)   '生産量の計算も行う
        Call 契約H更新(db, rsH, lngRoot設計ID, cur生産量反月, cur生産量K月)                              '生産量の更新
        rs.MoveNext
    Loop
    rsH社内備考.Close
    Set rsH社内備考 = Nothing
    rsH.Close
    Set rsH = Nothing
    rs.Close
    Set rs = Nothing
    If cnt > 0 Then
        If MsgBox(cnt & "件のルート設計に影響が有ります。" & vbCrLf & "連絡メールを送信しますか？", vbYesNo, "確認メッセージ") = vbYes Then
            Dim rt
'''''''''''''''''''''            rt = SendMail("mitsukawa.com:587", "rootdesignmodify@mitsukawa.com" & vbTab & "bcc" & vbTab & pubsMail, pubsMail & vbTab & "send:0120219969", "設計変更連絡　品番[" & str生機品番 & "](" & Format(Date, "yyyy/mm/dd") & ")", str社内備考メモ, "")
        End If
    End If
End Sub

Private Sub 契約B更新と生産量計算(db As DAO.Database, lng暫定設計ID As Long, lngRoot設計ID As Long, rsH As DAO.Recordset, ByRef cur生産量反月 As Currency, ByRef cur生産量K月 As Currency)
    Rem 生産量計算
    Dim W規格性量_生機長_インチ
    Dim WK_KYUSI_TBL(1 To 99) As Integer
    Dim WK_KYUSI_NO(1 To 6, 1 To 99) As Integer
    Dim WK_KYUSI_HON(1 To 6) As Integer
    Dim WK_MIKAKE_SENNDO(1 To 6) As Integer
    Dim WK_LOOP1(1 To 6) As Integer
    Dim WK_LOOP2(1 To 6) As Integer
    Dim WK_KYUSI_KUTI_ALL As Integer
    Dim WK生産量K月 As Double
    Dim l As Integer
    Dim i As Integer
    Erase WK_KYUSI_TBL
    Erase WK_KYUSI_NO
    Erase WK_KYUSI_HON
    Erase WK_MIKAKE_SENNDO
    Erase WK_LOOP1
    Erase WK_LOOP2
    WK生産量K月 = 0
    WK_KYUSI_KUTI_ALL = 0

    db.Execute "DELETE * FROM 設計Ｂ Where 設計ID = " & Format(lngRoot設計ID) & ";"
    Dim rsB As DAO.Recordset
    Set rsB = db.OpenRecordset("SELECT * FROM 設計Ｂ WHERE 設計ID=" & lng暫定設計ID & ";")
    Dim rsBroot As DAO.Recordset
    Set rsBroot = db.OpenRecordset("設計Ｂ")
    Do Until rsB.EOF
        i = i + 1
        If VarType(rsB![ループ長1]) = vbCurrency Then
            WK_LOOP1(i) = rsB![ループ長1]
        End If
        If VarType(rsB![ループ長2]) = vbCurrency Then
            WK_LOOP2(i) = rsB![ループ長2]
        End If
        If VarType(rsB![給糸本数]) = vbByte Then
            WK_KYUSI_HON(i) = rsB![給糸本数]
        End If
        If VarType(rsB![見掛繊度]) = vbCurrency Then
            WK_MIKAKE_SENNDO(i) = rsB![見掛繊度]
        End If
        If VarType(rsB![給糸NOU01]) = vbCurrency Then
            WK_KYUSI_NO(i, 1) = rsB![給糸NOU01]
        End If
        If VarType(rsB![給糸NOU02]) = vbCurrency Then
            WK_KYUSI_NO(i, 2) = rsB![給糸NOU02]
        End If
        If VarType(rsB![給糸NOU03]) = vbCurrency Then
            WK_KYUSI_NO(i, 3) = rsB![給糸NOU03]
        End If
        If VarType(rsB![給糸NOU04]) = vbCurrency Then
            WK_KYUSI_NO(i, 4) = rsB![給糸NOU04]
        End If
        If VarType(rsB![給糸NOU05]) = vbCurrency Then
            WK_KYUSI_NO(i, 5) = rsB![給糸NOU05]
        End If
        If VarType(rsB![給糸NOU06]) = vbCurrency Then
            WK_KYUSI_NO(i, 6) = rsB![給糸NOU06]
        End If
        If VarType(rsB![給糸NOU07]) = vbCurrency Then
            WK_KYUSI_NO(i, 7) = rsB![給糸NOU07]
        End If
        If VarType(rsB![給糸NOU08]) = vbCurrency Then
            WK_KYUSI_NO(i, 8) = rsB![給糸NOU08]
        End If
        If VarType(rsB![給糸NOD01]) = vbCurrency Then
            WK_KYUSI_NO(i, 9) = rsB![給糸NOD01]
        End If
        If VarType(rsB![給糸NOD02]) = vbCurrency Then
            WK_KYUSI_NO(i, 10) = rsB![給糸NOD02]
        End If
        If VarType(rsB![給糸NOD03]) = vbCurrency Then
            WK_KYUSI_NO(i, 11) = rsB![給糸NOD03]
        End If
        If VarType(rsB![給糸NOD04]) = vbCurrency Then
            WK_KYUSI_NO(i, 12) = rsB![給糸NOD04]
        End If
        If VarType(rsB![給糸NOD05]) = vbCurrency Then
            WK_KYUSI_NO(i, 13) = rsB![給糸NOD05]
        End If
        If VarType(rsB![給糸NOD06]) = vbCurrency Then
            WK_KYUSI_NO(i, 14) = rsB![給糸NOD06]
        End If
        If VarType(rsB![給糸NOD07]) = vbCurrency Then
            WK_KYUSI_NO(i, 15) = rsB![給糸NOD07]
        End If
        If VarType(rsB![給糸NOD08]) = vbCurrency Then
            WK_KYUSI_NO(i, 16) = rsB![給糸NOD08]
        End If

        rsBroot.AddNew
        rsBroot!設計ID = lngRoot設計ID
        rsBroot!設計明細NO = rsB!設計明細NO
        rsBroot!原糸区分 = rsB!原糸区分
        rsBroot!原糸品番 = rsB!原糸品番
        rsBroot!原糸コメント = rsB!原糸コメント
        rsBroot!給糸本数 = rsB!給糸本数
        rsBroot!見掛繊度 = rsB!見掛繊度
        rsBroot!給糸NOU01 = rsB!給糸NOU01
        rsBroot!給糸NOU02 = rsB!給糸NOU02
        rsBroot!給糸NOU03 = rsB!給糸NOU03
        rsBroot!給糸NOU04 = rsB!給糸NOU04
        rsBroot!給糸NOU05 = rsB!給糸NOU05
        rsBroot!給糸NOU06 = rsB!給糸NOU06
        rsBroot!給糸NOU07 = rsB!給糸NOU07
        rsBroot!給糸NOU08 = rsB!給糸NOU08
        rsBroot!給糸NOD01 = rsB!給糸NOD01
        rsBroot!給糸NOD02 = rsB!給糸NOD02
        rsBroot!給糸NOD03 = rsB!給糸NOD03
        rsBroot!給糸NOD04 = rsB!給糸NOD04
        rsBroot!給糸NOD05 = rsB!給糸NOD05
        rsBroot!給糸NOD06 = rsB!給糸NOD06
        rsBroot!給糸NOD07 = rsB!給糸NOD07
        rsBroot!給糸NOD08 = rsB!給糸NOD08
        rsBroot!ループ長1 = rsB!ループ長1
        rsBroot!ループ長2 = rsB!ループ長2
        rsBroot!交編率 = rsB!交編率
        rsBroot!QR原糸区分 = rsB!QR原糸区分
        rsBroot!QR原糸品名 = rsB!QR原糸品名
        rsBroot!QR原糸品種 = rsB!QR原糸品種
        rsBroot!QR糸加工コメント = rsB!QR糸加工コメント
        rsBroot!QR見掛繊度 = rsB!QR見掛繊度
        rsBroot!QR正味糸量 = rsB!QR正味糸量
        rsBroot!QR糸量単位 = rsB!QR糸量単位
        rsBroot!QR持掛糸量 = rsB!QR持掛糸量
        rsBroot!QR交編率 = rsB!QR交編率
        rsBroot!QR編立ロス率 = rsB!QR編立ロス率
        rsBroot!QR整経本数 = rsB!QR整経本数
        rsBroot!一周長1 = rsB!一周長1
        rsBroot!一周長2 = rsB!一周長2
        rsBroot.Update
        rsB.MoveNext
    Loop
    rsBroot.Close
    Set rsBroot = Nothing

    For i = 1 To 6
        For l = 1 To 16
            If WK_KYUSI_NO(i, l) <> 0 Then
                WK_KYUSI_TBL(WK_KYUSI_NO(i, l)) = WK_KYUSI_TBL(WK_KYUSI_NO(i, l)) + 1
                If WK_KYUSI_TBL(WK_KYUSI_NO(i, l)) = 1 Then
                    WK_KYUSI_KUTI_ALL = WK_KYUSI_KUTI_ALL + 1
                End If
            End If
        Next l
    Next i

    Dim WK_KYUSI_KUTI_UP As Integer
    Dim WK_KYUSI_KUTI_DOWN As Integer
    Dim WK_1REP_HON As Double
    Dim WK_1KAITEN_KG1 As Double
    Dim WK_1KAITEN_KG2 As Double
    Dim WK_1KAITEN_KG3 As Double
    Dim WK_GETUSAN_KG1 As Double
    Dim WK_GETUSAN_KG2 As Double
    WK_KYUSI_KUTI_UP = 0
    WK_KYUSI_KUTI_DOWN = 0
    WK_1REP_HON = 0
    WK_1KAITEN_KG1 = 0
    WK_1KAITEN_KG2 = 0
    WK_1KAITEN_KG3 = 0
    WK_GETUSAN_KG1 = 0
    WK_GETUSAN_KG2 = 0

    For i = 1 To 6
        WK_KYUSI_KUTI_UP = 0
        WK_KYUSI_KUTI_DOWN = 0
        For l = 1 To 16
            If WK_KYUSI_NO(i, l) <> 0 Then
                If l <= 8 Or WK_LOOP2(i) = 0 Then
                    WK_KYUSI_KUTI_UP = WK_KYUSI_KUTI_UP + 1
                Else
                    WK_KYUSI_KUTI_DOWN = WK_KYUSI_KUTI_DOWN + 1
                End If
            End If
        Next l
        If rsH!ウエル数 = 0 Or WK_KYUSI_KUTI_ALL = 0 Or WK_KYUSI_KUTI_UP = 0 Then
        Else
            WK_1REP_HON = Fix(((rsH!給糸本数 / WK_KYUSI_KUTI_ALL) * WK_KYUSI_KUTI_UP * WK_KYUSI_HON(i)) * 10000) / 10000
            WK_1KAITEN_KG1 = Fix(((WK_LOOP1(i) / 1000) * (rsH!編機針数 / rsH!ウエル数)) * 10000) / 10000
            WK_1KAITEN_KG2 = Fix((WK_1KAITEN_KG1 * WK_1REP_HON) * 10000) / 10000
            WK_1KAITEN_KG3 = Fix((WK_1KAITEN_KG2 * WK_MIKAKE_SENNDO(i)) * 10000) / 10000
            WK_GETUSAN_KG1 = Fix((WK_1KAITEN_KG3 * rsH!RPM * 0.0036) * 10000) / 10000
            WK_GETUSAN_KG2 = Fix((WK_GETUSAN_KG1 * (rsH!稼動率 / 100)) * 10000) / 10000
            WK生産量K月 = Fix(WK生産量K月 + WK_GETUSAN_KG2)
        End If
        If rsH!ウエル数 = 0 Or WK_KYUSI_KUTI_ALL = 0 Or WK_KYUSI_KUTI_DOWN = 0 Then
        Else
            WK_1REP_HON = Fix(((rsH!給糸本数 / WK_KYUSI_KUTI_ALL) * WK_KYUSI_KUTI_DOWN * WK_KYUSI_HON(i)) * 10000) / 10000
            WK_1KAITEN_KG1 = Fix(((WK_LOOP2(i) / 1000) * (rsH!編機針数 / rsH!ウエル数)) * 10000) / 10000
            WK_1KAITEN_KG2 = Fix((WK_1KAITEN_KG1 * WK_1REP_HON) * 10000) / 10000
            WK_1KAITEN_KG3 = Fix((WK_1KAITEN_KG2 * WK_MIKAKE_SENNDO(i)) * 10000) / 10000
            WK_GETUSAN_KG1 = Fix((WK_1KAITEN_KG3 * rsH!RPM * 0.0036) * 10000) / 10000
            WK_GETUSAN_KG2 = Fix((WK_GETUSAN_KG1 * (rsH!稼動率 / 100)) * 10000) / 10000
            WK生産量K月 = Fix(WK生産量K月 + WK_GETUSAN_KG2)
        End If
    Next i

    If rsH!規格性量_生機重量 = 0 Then
        cur生産量反月 = 0
    Else
        cur生産量反月 = Fix(WK生産量K月 / rsH!規格性量_生機重量)
    End If

    cur生産量K月 = WK生産量K月
End Sub
