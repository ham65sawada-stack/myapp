Private Sub Just_Click()
    Dim dial As String
    dial = Trim(Nz(Me.検索dial))

    Dim dial_len As Integer
    dial_len = Len(dial)

    Dim cylinder As String
    cylinder = Trim(Nz(Me.検索cylinder))

    Dim cylinder_len As Integer
    cylinder_len = Len(cylinder)

    If dial_len = 0 And cylinder_len = 0 Then
        Me.Requery
        Me.Filter = "[dial] LIKE '*'"
        Me.FilterOn = True
        Me.FilterOn = False
        Exit Sub
    End If

    Dim WK_WHERE As String
    WK_WHERE = ""

    If dial_len > 0 Then
        WK_WHERE = WK_WHERE & "([dial] = '" & dial & "'"
        WK_WHERE = WK_WHERE & " OR [dial] = '" & dial & "@')"
    End If

    If cylinder_len > 0 Then
        If WK_WHERE <> "" Then
            WK_WHERE = WK_WHERE & " AND "
        End If

        WK_WHERE = WK_WHERE & "([cylinder] = '" & cylinder & "'"
        WK_WHERE = WK_WHERE & " OR [cylinder] = '" & cylinder & "@')"
    End If
    Call set_sql(WK_WHERE)
End Sub

Private Sub set_sql(WK_WHERE As String)
    Dim S As String
    S = "SELECT 組織CD, 組織図, dial, cylinder FROM 組織図マスタ WHERE 組織CD > Forms!設計Ｈメンテ!出合い*100000 And 組織CD < (Forms!設計Ｈメンテ!出合い+1)*100000"
    If WK_WHERE <> "" Then
        S = S & " AND (" & WK_WHERE & ")"
    End If
    S = S & " ORDER BY 組織CD DESC;"
    Me.RecordSource = S
End Sub

Private Sub serial_Click()
    Dim dial As String
    dial = Trim(Nz(Me.検索dial))

    Dim dial_len As Integer
    dial_len = Len(dial)

    Dim cylinder As String
    cylinder = Trim(Nz(Me.検索cylinder))

    Dim cylinder_len As Integer
    cylinder_len = Len(cylinder)

    If dial_len = 0 And cylinder_len = 0 Then
        Me.Requery
        Me.Filter = "[dial] LIKE '*'"
        Me.FilterOn = True
        Me.FilterOn = False
        Exit Sub
    End If

    Dim lcm_Long As Long
    lcm_Long = LCM(dial_len, cylinder_len)
    If lcm_Long = 0 Then
        If dial_len > 0 Then
            lcm_Long = dial_len
        Else
            lcm_Long = cylinder_len
        End If
    End If

    Dim i As Integer
    If dial_len > 0 Then
        Dim dial_Multiple As Long
        dial_Multiple = lcm_Long / dial_len
        For i = 1 To dial_Multiple
            If i > 1 Then
                dial = dial & dial
            End If
        Next i
    End If

    If cylinder_len > 0 Then
        Dim cylinder_Multiple As Long
        cylinder_Multiple = lcm_Long / cylinder_len
        For i = 1 To cylinder_Multiple
            If i > 1 Then
                cylinder = cylinder & cylinder
            End If
        Next i
    End If

    Dim WK_WHERE As String
    WK_WHERE = ""

    For i = 1 To lcm_Long
        If i > 1 Then
            WK_WHERE = WK_WHERE & " OR "
        End If

        WK_WHERE = WK_WHERE & "("

        If dial_len > 0 Then
            dial = RotateStringLeft(dial)
            WK_WHERE = WK_WHERE & "([dial] = '" & Left(dial, dial_len) & "'"
            WK_WHERE = WK_WHERE & " OR [dial] = '" & Left(dial, dial_len) & "@')"
            If cylinder_len > 0 Then
                WK_WHERE = WK_WHERE & " AND "
            End If
        End If
        If cylinder_len > 0 Then
            cylinder = RotateStringLeft(cylinder)
            WK_WHERE = WK_WHERE & "([cylinder] = '" & Left(cylinder, cylinder_len) & "'"
            WK_WHERE = WK_WHERE & " OR [cylinder] = '" & Left(cylinder, cylinder_len) & "@')"
        End If
        WK_WHERE = WK_WHERE & ")"
    Next i
    Call set_sql(WK_WHERE)
End Sub

''' <summary>
''' 文字列を左に1文字ローテーションします。
''' </summary>
''' <param name="inputText">対象の文字列</param>
''' <returns>ローテーション後の文字列</returns>
Private Function RotateStringLeft(ByVal inputText As String) As String
    ' 文字列が1文字以下の場合は、そのまま返す
    If Len(inputText) <= 1 Then
        RotateStringLeft = inputText
        Exit Function
    End If

    ' 2文字目から最後までを取得し、それに最初の1文字を連結する
    RotateStringLeft = Mid(inputText, 2) & Left(inputText, 1)
End Function

''' <summary>
''' 2つの整数の最小公倍数 (LCM) を計算します。
''' </summary>
''' <param name="a">1つ目の整数</param>
''' <param name="b">2つ目の整数</param>
''' <returns>最小公倍数</returns>
Private Function LCM(ByVal a As Long, ByVal b As Long) As Long
    ' 0が含まれる場合のLCMは0とする
    If a = 0 Or b = 0 Then
        LCM = 0
        Exit Function
    End If

    ' 公式: LCM(a, b) = |a * b| / GCD(a, b)
    ' 大きな数でのオーバーフローを防ぐため、先に割り算を行う
    LCM = (Abs(a) / GCD(a, b)) * Abs(b)
End Function

''' <summary>
''' 2つの整数の最大公約数 (GCD) を計算します。（ユークリッドの互除法）
''' この関数はLCM関数から呼び出されます。
''' </summary>
''' <param name="a">1つ目の整数</param>
''' <param name="b">2つ目の整数</param>
''' <returns>最大公約数</returns>
Private Function GCD(ByVal a As Long, ByVal b As Long) As Long
    Dim temp As Long

    ' 正の数で計算を行う
    a = Abs(a)
    b = Abs(b)

    ' ユークリッドの互除法
    Do While b <> 0
        temp = b
        b = a Mod b
        a = temp
    Loop

    GCD = a
End Function
