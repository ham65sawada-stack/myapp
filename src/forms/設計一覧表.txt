Private Function 採番() As Long
    Dim db As DAO.Database
    Dim rs As DAO.Recordset
    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT カウント FROM 設計ＣＴマスタ where コード='01'")
    If Not rs.EOF Then
        rs.Edit
            採番 = rs.Fields("カウント")
            rs.Fields("カウント") = rs.Fields("カウント") + 1
        rs.Update
    Else
        採番 = 0
    End If
    rs.Close
    Set rs = Nothing
    Set db = Nothing
End Function

Private Sub 社内備考検索()
    Dim wd As String
    wd = Trim(IfString(Me.検索社内備考))
    Dim st, en As Currency
    st = val(IfNumeric(Me.検索開始社内備考))
    en = val(IfNumeric(Me.検索終了社内備考))

    Dim rs As DAO.Recordset
    Set rs = CurrentDb.OpenRecordset("WK設計一覧表")

    Do While Not rs.EOF()
        Dim S As String
        S = DLookup("社内備考メモ", "設計Ｈ社内備考", "設計ID=" & IfNumeric(rs!設計ID))
        Dim p As Long
        p = InStr(1, S, wd, vbTextCompare)
        Dim valstr As String
        valstr = ""
        If 0 < p Then
            p = p + Len(wd)
            Dim l As Long
            l = Len(S)
            Dim i As Long
            i = 6
            Do While 0 < i
                If l < p Then
                    i = 0
                Else
                    Dim ps As String
                    ps = Mid(S, p, 1)
                    If ps = "0" Or ps = "1" Or ps = "2" Or ps = "3" Or ps = "4" Or ps = "5" Or ps = "6" Or ps = "7" Or ps = "8" Or ps = "9" Or ps = "." Then
                        valstr = valstr & ps
                        p = p + 1
                        i = i - 1
                    Else
                        i = 0
                    End If
                End If
            Loop
        End If
        Dim valnum As Currency
        valnum = val(valstr)
        Debug.Print S
        Debug.Print valnum
        If en = 0 Then
            If valnum < st Then
                rs.Delete
            End If
        Else
            If valnum < st Or en < valnum Then
                rs.Delete
            End If
        End If
        rs.MoveNext
    Loop

    rs.Close
    Set rs = Nothing
End Sub
