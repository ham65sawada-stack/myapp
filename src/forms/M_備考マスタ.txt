Private Sub Form_BeforeUpdate(Cancel As Integer)
    If VarType(Me![備考]) = vbString Then
        If Trim(Me![備考]) = "" Then
            MsgBox "備考を入力して下さい", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
            Cancel = True
        End If
    Else
        MsgBox "備考を入力して下さい", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Cancel = True
    End If
End Sub

Private Sub 閉じる_Click()
On Error GoTo Err_閉じる_Click
'    DoCmd.DoMenuItem acFormBar, acRecordsMenu, acSaveRecord, , acMenuVer70
    DoCmd.RunCommand acCmdSaveRecord
Exit_閉じる_Click:
    DoCmd.Close
    Exit Sub

Err_閉じる_Click:
    Resume Exit_閉じる_Click
End Sub

Private Sub Form_Delete(Cancel As Integer)
On Error GoTo Err_Form_Delete
    Dim Anser As Integer
    Anser = MsgBox("削除します確認して下さい。", vbYesNo + vbCritical + vbDefaultButton2, "確認メッセージ")
    If Anser = vbNo Then
        Cancel = True
    End If
Exit_Form_Delete:
    Exit Sub
Err_Form_Delete:
    Resume Exit_Form_Delete
End Sub

Private Sub Form_Error(DataErr As Integer, Response As Integer)
    If DataErr = 3022 Then
        MsgBox "同じデータが入力されています", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Response = True
    End If
End Sub
