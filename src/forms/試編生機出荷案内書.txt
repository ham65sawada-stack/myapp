Private Sub Form_Current()
    Dim db As Database
    Dim rs As Recordset
    Dim vKensuu As Variant
    Dim lCnt As Long
    Set db = CurrentDb
    lCnt = 0
    Set rs = db.OpenRecordset("SELECT 原糸品番,SUM(交編率) AS Sum交編率 FROM 設計Ｂ where 設計ＩＤ=" & Me.設計NO & "AND 原糸区分<>'X' GROUP BY 原糸区分,原糸品番")
    Do While Not rs.EOF
        lCnt = lCnt + 1
        Me("原糸品番" & lCnt) = Trim(rs.Fields("原糸品番"))
        Me("交編率" & lCnt) = Trim(rs.Fields("Sum交編率"))
        rs.MoveNext
    Loop
    rs.Close
    Set rs = Nothing
    db.Close
    Set db = Nothing
    If IsNull(Me.写) Or Me.写 = "" Then
        Call 写し作成
    End If
    vKensuu = dcount("*", "試編出荷データ", "設計ＮＯ=" & Me.設計NO)
    If IsNumeric(vKensuu) And vKensuu > 0 Then
        Me.修正.Enabled = True
    Else
        Me.修正.Enabled = False
    End If
    Call 罰計計算(1)
End Sub

Private Function 更新() As Boolean
    DoCmd.RunCommand acCmdSaveRecord
    Dim db As DAO.Database
    Dim qrdef As QueryDef
    更新 = False
    If MsgBox("更新します確認して下さい。", vbYesNoCancel + vbCritical + vbDefaultButton3, "確認メッセージ") = vbYes Then
        If IsNull(Me.メーカ) Then
            MsgBox "メーカーを入力して下さい。", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
            Exit Function
        End If
        If Not IsDate(Me.伝票日付) Then
            MsgBox "伝票日付を入力して下さい。", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
            Exit Function
        End If

        Call 試編出荷データワークの属性チェック(Me)
        DoCmd.RunCommand acCmdSaveRecord

        Set db = CurrentDb
        db.Execute "DELETE * FROM 試編出荷データ Where 連番=" & Format(Me.連番)
        db.Execute "INSERT INTO 試編出荷データ SELECT * FROM 試編出荷データワーク"
        db.Close
        更新 = True
    End If
End Function

Private Sub 目付計算(NO As Double)
    If Not IsNumeric(Me("長" & NO)) Then Me("長" & NO) = 0
    If Not IsNumeric(Me("重量" & NO)) Then Me("重量" & NO) = 0
    If Not IsNumeric(Me("目付" & NO)) Then Me("目付" & NO) = 0
    If (IsNull(Me("品番" & NO)) Or Trim(Me("品番" & NO)) = "") And _
       (IsNull(Me("反番" & NO)) Or Trim(Me("反番" & NO)) = "") Then
        Me("長" & NO) = 0
        Me("重量" & NO) = 0
        Me("目付" & NO) = 0
    Else
        If NO > 1 Then
            If Not IsNumeric(Me("巾" & NO)) Then Me("巾" & NO) = IIf(IsNumeric(Me("巾" & NO - 1)), Me("巾" & NO - 1), 0)
            If Me("巾" & NO) = 0 Then Me("巾" & NO) = IIf(IsNumeric(Me("巾" & NO - 1)), Me("巾" & NO - 1), 0)
            If Not IsNumeric(Me("目付" & NO)) Then Me("目付" & NO) = IIf(IsNumeric(Me("目付" & NO - 1)), Me("目付" & NO - 1), 0)
            If Me("目付" & NO) = 0 Then Me("目付" & NO) = IIf(IsNumeric(Me("目付" & NO - 1)), Me("目付" & NO - 1), 0)
        End If
        If Me("重量" & NO) = 0 Then
            Me("重量" & NO) = Format(Me("長" & NO) * Me("目付" & NO) / 1000, "0.00")
        End If
        If Me("長" & NO) = 0 And Me("目付" & NO) <> 0 Then
            Me("長" & NO) = Format(Me("重量" & NO) / Me("目付" & NO) * 1000, "0.0")
        End If
        If Me("目付" & NO) = 0 And Me("長" & NO) <> 0 Then
            Me("目付" & NO) = Format((Me("重量" & NO) * 1000) / Me("長" & NO), "0")
        End If
    End If
End Sub

Private Sub 罰計計算(NO As Double)
    Dim i As Integer
    Dim g As Integer
    g = 0
    For i = 1 To 10
        g = g + IIf(IsNumeric(Me("落傷" & i)), Me("落傷" & i), 0) + IIf(IsNumeric(Me("針折" & i)), Me("針折" & i), 0) + IIf(IsNumeric(Me("ハネ" & i)), Me("ハネ" & i), 0) + IIf(IsNumeric(Me("飛込" & i)), Me("飛込" & i), 0) + IIf(IsNumeric(Me("汚れ" & i)), Me("汚れ" & i), 0) + IIf(IsNumeric(Me("その他" & i)), Me("その他" & i), 0)
    Next i
    Me.非連結罰計計 = g
End Sub

Private Sub Form_KeyDown(KeyCode As Integer, Shift As Integer)
    If Shift = 0 Then
        Select Case KeyCode
        Case 112    'F1
            KeyCode = 0
            Call 取消終了_Click
        Case 113    'F2
            KeyCode = 0
            Call 印刷0枚_Click
        Case 114    'F3
            KeyCode = 0
            Call 印刷1枚_Click
        Case 115    'F4
            KeyCode = 0
            Call 印刷2枚_Click
        Case 116    'F5
            KeyCode = 0
            Call 印刷3枚_Click
        Case 27     'ESC
            KeyCode = 0
            If Me.Dirty Then
                On Error Resume Next
                DoCmd.RunCommand acCmdUndo
                On Error GoTo 0
                If Not Me.Dirty Then
                    Call Form_Current
                End If
            End If
        End Select
    End If
    If Shift = 1 Then
        Select Case KeyCode
        Case 112    '+F1
            KeyCode = 0
            If Me.修正.Enabled Then
                Call 修正_Click
            End If
        Case 113    '+F2
            KeyCode = 0
            Call 削除_Click
        Case 114    '+F3
            KeyCode = 0
        Case 115    '+F4
            KeyCode = 0
        Case 116    '+F5
            KeyCode = 0
        End Select
    End If
End Sub
