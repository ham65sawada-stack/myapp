Option Compare Database
Dim vOyaForm As Variant

Private Sub Form_Open(Cancel As Integer)
    On Error Resume Next
    vOyaForm = Me.OpenArgs
    On Error GoTo 0
End Sub

Private Sub 取り消し_Click()
    DoCmd.OpenForm "試編生機出荷案内書", , , , , , vOyaForm
    DoCmd.Close acForm, Me.Name
End Sub

Private Sub 選択_Click()
    On Error GoTo Err_選択_Click
    If IsNumeric(Me.連番) Then
        Dim db As DAO.Database
        Set db = CurrentDb
        db.Execute "DELETE 試編出荷データワーク.* FROM 試編出荷データワーク"
        db.Execute "INSERT INTO 試編出荷データワーク SELECT 試編出荷データ.* FROM 試編出荷データ Where 連番=" & Me.連番
        db.Close
        DoCmd.OpenForm "試編生機出荷案内書", , , , , , vOyaForm
        DoCmd.Close acForm, Me.Name
    End If
Exit_選択_Click:
    Exit Sub
Err_選択_Click:
    Exit Sub
End Sub

Private Sub Form_KeyDown(KeyCode As Integer, Shift As Integer)
    If Shift = 0 Then
        Select Case KeyCode
        Case 112    'F1
            KeyCode = 0
            Call 選択_Click
        Case 113    'F2
            KeyCode = 0
        Case 114    'F3
            KeyCode = 0
        Case 115    'F4
            KeyCode = 0
        Case 116    'F5
            KeyCode = 0
            Call 取り消し_Click
        Case 27     'ESC
            KeyCode = 0
            If Me.Dirty Then
                On Error Resume Next
                DoCmd.RunCommand acCmdUndo
                On Error GoTo 0
                If Not Me.Dirty Then
'                   Call Form_Current
                End If
            End If
        End Select
    End If
End Sub
