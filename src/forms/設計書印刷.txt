Private Sub Form_Open(Cancel As Integer)
    If Trim(Me![編立工場CD]) = "MA79" Then
        Me![画面編立工場ＣＤ] = Null
        Me![画面編立工場名] = Null
        Me![画面号機] = Me![号機]
    Else
        Me![画面編立工場ＣＤ] = Trim(Me![編立工場CD])
        Me![画面編立工場名] = Me![編立工場名]
        Me![MailAdd] = [画面編立工場ＣＤ].[Column](3)
        Me![画面号機] = Null
    End If
End Sub
