Private Sub OK_Click()
    Dim db       As DAO.Database
    Dim rs       As DAO.Recordset
    Dim SampleNo As String
    SampleNo = IIf(IsNull(Me.分類), Space(3), Left(Trim(Me.分類) & Space(3), 3)) & Format(IIf(IsNumeric(Me.ｻﾝﾌﾟﾙ№), Me.ｻﾝﾌﾟﾙ№, 0), "0000")
    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT ｻﾝﾌﾟﾙ№,ＱＲ修正区分 FROM 設計Ｈ where 設計ID=" & Me.設計ID)
    If Not rs.EOF Then
        rs.Edit
            rs.Fields("ｻﾝﾌﾟﾙ№") = SampleNo
            rs.Fields("ＱＲ修正区分") = "*"
        rs.Update
    End If
    rs.Close
    Set rs = Nothing
    Set db = Nothing
    DoCmd.Close
End Sub

Private Sub 採番_Click()
    If MsgBox("サンプル№を採番します確認して下さい。", vbYesNo + vbCritical + vbDefaultButton2, "確認メッセージ") = vbYes Then
        ｻﾝﾌﾟﾙ№ = サンプル№採番()
    End If
End Sub

Private Function サンプル№採番() As Long
    Dim db    As DAO.Database
    Dim rs    As DAO.Recordset
    Dim Kubun As String

    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT max(サンプル№) AS maxサンプル№ FROM 設計Ｈ where LEFT(サンプル№,3)='" & IIf(IsNull(Me.分類), Space(3), Left(Trim(Me.分類) & Space(3), 3)) & "'")
    If Not rs.EOF Then
        If IsNull(rs.Fields("maxサンプル№")) Then
            サンプル№採番 = 1
        Else
            サンプル№採番 = val(Mid(Trim(rs.Fields("maxサンプル№")), 4)) + 1
        End If
    Else
        サンプル№採番 = 0
    End If
    rs.Close
    Set rs = Nothing
    Set db = Nothing
End Function
