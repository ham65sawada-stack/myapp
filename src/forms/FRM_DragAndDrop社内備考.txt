Private Function CopyFile(コピー元 As String) As Boolean
    sコピー元PDFの新フルパス名 = ""
    If InStr(1, コピー元, ".", vbTextCompare) > 0 Then
        Select Case Right(コピー元, 4)
            Case ".PDF"
                sコピー元PDF名 = Mid(コピー元, InStrRev(コピー元, "\", -1, vbTextCompare) + 1)
                sコピー元PDF名 = Choose(val(Right(f, 1)), "①", "②", "③", "④", "⑤", "⑥", "⑦", "⑧", "⑨", "⑩", "⑪", "⑫", "⑬", "⑭", "⑮") & Mid(sコピー元PDF名, 1, Len(sコピー元PDF名) - 4)
                sコピー元PDFの新フルパス名 = "C:\pin\Mitsukawa\Sekkei\PDF\" & val(IfNumeric(Form_設計Hメンテ.設計ID)) & sコピー元PDF名 & ".pdf"
            Case Else
                CopyFile = False
                Exit Function
        End Select
        If s書き込み先PDFのフルパス名 <> "" And Dir(s書き込み先PDFのフルパス名) <> "" Then
            Dim Anser As Variant
            Anser = MsgBox("追加しますか？" & vbCrLf & "[はい]追加" & vbCrLf & "[いいえ]置き換え" & vbCrLf & "[キャンセル]何もしない", vbYesNoCancel, "確認メッセージ")
            If Anser = vbYes Then
                If PDFTK_File結合(コピー元, s書き込み先PDFのフルパス名) = False Then
                    MsgBox "指定ファイルの結合に失敗しました。", , "エラーメッセージ"
                    Exit Function
                End If
                CopyFile = True
            ElseIf Anser = vbNo Then
                On Error Resume Next
                Kill s書き込み先PDFのフルパス名
                FileCopy コピー元, s書き込み先PDFのフルパス名
                If Dir(s書き込み先PDFのフルパス名) = "" Then
                    MsgBox "指定ファイルの結合に失敗しました。", , "エラーメッセージ"
                    Exit Function
                Else
                    Form_設計Hメンテ(f) = sコピー元PDF名
                End If
                On Error GoTo 0
                CopyFile = True
            Else
                CopyFile = False
                Exit Function
            End If
        Else
            On Error Resume Next
            FileCopy コピー元, s書き込み先PDFのフルパス名
            If Dir(s書き込み先PDFのフルパス名) = "" Then
                MsgBox "指定ファイルの結合に失敗しました。", , "エラーメッセージ"
                Exit Function
            Else
                Form_設計Hメンテ(f) = sコピー元PDF名
            End If
            On Error GoTo 0
            CopyFile = True
        End If
    End If
End Function

'--- PDFファイルの結合 ---
'引数①:元ファイル FullPath
'引数②:追加ファイル FullPath
Private Function PDFTK_File結合(元ファイル As String, 追加ファイル As String) As Boolean
On Error GoTo Err_PDFTK_File結合
    Dim strCommand As String
    '--①ファイルの名前換え（漢字が使えないおそれがあるため）--
    FileCopy 元ファイル, CurrentProject.path & "\xyz_aaa.pdf"
    FileCopy 追加ファイル, CurrentProject.path & "\xyz_bbb.pdf"

    '-- 作業フォルダへの移動 --
    ChDir CurrentProject.path

    '--②PDFTKによるファイルの結合 --
    strCommand = "pdftk.exe xyz_aaa.pdf xyz_bbb.pdf cat output xyz_ccc.pdf"

    ' WSHを使ってコマンドを実行する
    Dim objWSH As Object              ' WSH
    Set objWSH = CreateObject("WScript.Shell")
    objWSH.Run strCommand, vbHide, True ' 第3引数がTrueなら、同期実行
    Set objWSH = Nothing

    '--③結合後ファイルの名前換え
    Kill 追加ファイル
    FileCopy "xyz_ccc.pdf", 追加ファイル

    '--④作業ファイルの削除
    Kill "xyz_aaa.pdf"
    Kill "xyz_bbb.pdf"
    Kill "xyz_ccc.pdf"

    PDFTK_File結合 = True
    Exit Function

Err_PDFTK_File結合:
    MsgBox Err.Number & "；" & Err.Description
    PDFTK_File結合 = False

End Function
