Private Sub OK_Click()
    Dim db       As DAO.Database
    Dim rs       As DAO.Recordset
    Dim Kubun    As String
    Dim HangerNo As String
    Dim H_No As Long
    Select Case 区分
    Case 1
        Kubun = "W"
    Case 2
        Kubun = "S"
    Case Else
        Kubun = ""
    End Select

    If IsNumeric(Me.ハンガー№) Then
        H_No = Me.ハンガー№
    Else
        H_No = 0
    End If

    If H_No > 9999 Then
        HangerNo = Kubun & Format(H_No, "00000")
    Else
        HangerNo = Kubun & Format(H_No, "0000")
    End If
    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT ハンガー№,ＱＲ修正区分 FROM 設計Ｈ where 設計ID=" & Me.設計ID)
    If Not rs.EOF Then
        rs.Edit
            rs.Fields("ハンガー№") = HangerNo
            rs.Fields("ＱＲ修正区分") = "*"
        rs.Update
    End If
    rs.Close
    Set rs = Nothing
    Set db = Nothing
    DoCmd.Close
End Sub

Private Sub 採番_Click()
    If MsgBox("ハンガー№を採番します確認して下さい。", vbYesNo + vbCritical + vbDefaultButton2, "確認メッセージ") = vbYes Then
        ハンガー№ = ハンガー№採番()
    End If
End Sub

Private Function ハンガー№採番() As Long
    Dim db    As DAO.Database
    Dim rs    As DAO.Recordset
    Dim Kubun As String
    Select Case 区分
    Case 1
        Kubun = "W"
    Case 2
        Kubun = "S"
    Case Else
        ハンガー№採番 = 0
        Exit Function
    End Select

    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT カウント FROM 設計ＣＴマスタ where コード='" & Kubun & "'")
    If Not rs.EOF Then
        rs.Edit
            ハンガー№採番 = rs.Fields("カウント")
            rs.Fields("カウント") = rs.Fields("カウント") + 1
        rs.Update
    Else
        ハンガー№採番 = 0
    End If
    rs.Close
    Set rs = Nothing
    Set db = Nothing
End Function
