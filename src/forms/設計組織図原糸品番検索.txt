Private Sub 絞込み_Click()
    Dim sSQL As String
    Dim sScanWord As String
    Dim sWord(30) As String
    Dim iCnt As Integer
    Dim i As Integer
    Dim iPosEnd As Integer

    If Not IsNull(Me.原糸品種絞り込み条件) Then
        sScanWord = Trim(Me.原糸品種絞り込み条件)
        If Len(sScanWord) > 0 Then
            iCnt = 0
            Do While Len(sScanWord) > 0
                iCnt = iCnt + 1
                iPosEnd = InStr(1, sScanWord, " ")
                If iPosEnd = 0 Then
                    sWord(iCnt) = "*" & sScanWord & "*"
                    sScanWord = ""
                Else
                    sWord(iCnt) = "*" & Trim(Left(sScanWord, iPosEnd)) & "*"
                    sScanWord = Trim(Mid(sScanWord, iPosEnd))
                End If
            Loop

            If iCnt > 0 Then
                sSQL = "SELECT IIf([設計区分]=1,'ル','試') AS 区分, 設計原糸一覧マスタ.原糸品番"
                sSQL = sSQL & " FROM 設計原糸一覧マスタ"
                sSQL = sSQL & " GROUP BY IIf([設計区分]=1,'ル','試'), 設計原糸一覧マスタ.原糸品番"
                For i = 1 To iCnt
                    If i = iCnt Then
                        If i = 1 Then
                            sSQL = sSQL & " HAVING 原糸品番 Like '" & sWord(i) & "'"
                        Else
                            sSQL = sSQL & " 原糸品番 Like '" & sWord(i) & "'"
                        End If
                    Else
                        If i = 1 Then
                            sSQL = sSQL & " HAVING 原糸品番 Like '" & sWord(i) & "' and"
                        Else
                            sSQL = sSQL & " 原糸品番 Like '" & sWord(i) & "' and"
                        End If
                    End If
                Next i
                sSQL = sSQL & ";"
                Me.Painting = False
                Me.絞込みリスト.RowSource = sSQL
            Else
                Me.Painting = False
                Me.絞込みリスト.RowSource = ""
            End If

            Me.絞込みリスト.Requery
            Me.Painting = True
        End If
    End If

    On Error Resume Next
    Me.原糸品種絞り込み条件.SetFocus
    On Error GoTo 0
End Sub
