Option Compare Database
Dim vOyaForm As Variant

Private Sub Cancel_Click()
    Me!PassWord = "<<Cancel>>"
    DoCmd.Close
End Sub

Private Sub Form_Open(Cancel As Integer)
    On Error Resume Next
    vOyaForm = Me.OpenArgs
'    Forms(vOyaForm).Visible = False
    Me.OrderByOn = False
    Me.FilterOn = False
    Me.OrderBy = ""
    Me.Filter = ""
    Me!PassWord = Forms(vOyaForm).Tag
    On Error GoTo 0
End Sub

Private Sub Form_Unload(Cancel As Integer)
    On Error Resume Next
    If IsNull(Me!PassWord) Then
        Forms(vOyaForm).Tag = ""
    Else
        Forms(vOyaForm).Tag = Me!PassWord
    End If
    On Error GoTo 0
End Sub

    DoCmd.Close
End Sub
