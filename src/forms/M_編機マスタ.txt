Private Sub Form_BeforeUpdate(Cancel As Integer)
    If Not VarType(Me![編機CD]) = vbLong Then
        MsgBox "編機ＣＤを入力して下さい", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Cancel = True
        Exit Sub
    End If
    If Not VarType(Me![編機メーカ名]) = vbString Then
        MsgBox "編機メーカＣＤを入力して下さい", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Cancel = True
        Exit Sub
    End If
End Sub

Private Sub Form_Error(DataErr As Integer, Response As Integer)
    If DataErr = 3022 Then
        MsgBox "同じデータが入力されています", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Response = True
    ElseIf DataErr = 3200 Then
        MsgBox "号機マスタに使用されている為、削除できません。", vbOKOnly + vbCritical + vbDefaultButton2, "エラーメッセージ"
        Response = True
    Else
        MsgBox Format(DataErr)
    End If
End Sub

Private Sub Form_Delete(Cancel As Integer)
On Error GoTo Err_Form_Delete
    Dim Anser As Integer
    Anser = MsgBox("削除します確認して下さい。", vbYesNo + vbCritical + vbDefaultButton2, "確認メッセージ")
    If Anser = vbNo Then
        Cancel = True
    End If
Exit_Form_Delete:
    Exit Sub
Err_Form_Delete:
    Resume Exit_Form_Delete
End Sub
