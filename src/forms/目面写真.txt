Dim s設計ID As String
Dim sモード As String


Private Sub Form_Open(Cancel As Integer)
    On Error Resume Next
    If val(IfNumeric(Form_設計Hメンテ!暫定設計ID)) > 0 Then
        s設計ID = Format(val(IfNumeric(Form_設計Hメンテ!暫定設計ID)), "000000")
    Else
        s設計ID = Format(val(IfNumeric(Form_設計Hメンテ.設計ID)), "000000")
    End If
    sモード = Form_設計Hメンテ.モード.Caption
    On Error GoTo 0
    Me.ListView表.OLEDropMode = 1
    Me.ListView表.View = 2
    Me.ListView表.Visible = False
    Me.ListView裏.OLEDropMode = 1
    Me.ListView裏.View = 2
    Me.ListView裏.Visible = False

    Dim FileName As String
    FileName = Dir(Path目面写真 & "表\" & s設計ID & ".*")
    If FileName = "" Then
        Me.表面.Visible = False
        Me.ListView表.Visible = True
        Me.表面削除.Enabled = False
    Else
        Me.表面.Picture = Path目面写真 & "表\" & FileName
        Me.表面.HyperlinkAddress = Me.表面.Picture
        Me.表面.Visible = True
        If val(IfNumeric(Form_設計Hメンテ!暫定設計ID)) = 0 Then
            Me.表面削除.Enabled = True
        End If
    End If
    FileName = Dir(Path目面写真 & "裏\" & s設計ID & ".*")
    If FileName = "" Then
        Me.裏面.Visible = False
        Me.ListView裏.Visible = True
        Me.裏面削除.Enabled = False
    Else
        Me.裏面.Picture = Path目面写真 & "裏\" & FileName
        Me.裏面.HyperlinkAddress = Me.裏面.Picture
        Me.裏面.Visible = True
        If val(IfNumeric(Form_設計Hメンテ!暫定設計ID)) = 0 Then
            Me.裏面削除.Enabled = True
        End If
    End If
End Sub

Private Sub 表面削除_Click()
    If val(IfNumeric(Form_設計Hメンテ!暫定設計ID)) = 0 Then
        If MsgBox("表面を削除しますか？", vbOKCancel, "確認メッセージ") = vbOK Then
            On Error Resume Next
            Me.閉じる.SetFocus
            On Error GoTo 0
            Me.ListView表.Visible = True
            Me.表面.Visible = False
            Me.表面削除.Enabled = False
            On Error Resume Next
            Kill Me.表面.Picture
            On Error GoTo 0
        End If
    End If
End Sub

Private Sub 閉じる_Click()
    DoCmd.Close acForm, Me.Name
End Sub

Private Sub ListView表_OLEDragDrop(Data As Object, Effect As Long, Button As Integer, Shift As Integer, x As Single, Y As Single)
    If val(IfNumeric(Form_設計Hメンテ!暫定設計ID)) = 0 Then
        Dim S As String
        On Error Resume Next
        S = ""
        S = Data.files(1)
        On Error GoTo 0
        If S <> "" Then
            Dim newFileName As String
            newFileName = CopyFile(S, "表")
            If newFileName <> "" Then
                On Error Resume Next
                Me.閉じる.SetFocus
                On Error GoTo 0
                Me.表面.Picture = newFileName
                Me.表面.HyperlinkAddress = newFileName
                Me.表面.Visible = True
                Me.ListView表.Visible = False
                Me.表面削除.Enabled = True
            End If
        End If
    End If
End Sub

Private Function CopyFile(コピー元 As String, 表面 As String) As String
    CopyFile = ""
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    Dim ext As String
    ext = fso.GetExtensionName(コピー元)
    If ext = "png" Or ext = "jpeg" Or ext = "jpg" Then
        On Error Resume Next
        Kill Path目面写真 & 表面 & "\" & s設計ID & ".*"
        On Error GoTo 0
        CopyFile = Path目面写真 & 表面 & "\" & s設計ID & "." & ext
        FileCopy コピー元, CopyFile
    End If
End Function
