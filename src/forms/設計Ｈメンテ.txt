        Else
            Me.機能選択リスト.Selected(i) = False
        End If
    Next i

    w特徴選択 = Me.特徴選択 & String(50, "0")
    For i = 0 To Me.特徴選択リスト.ListCount - 1

            Me.特徴選択リスト.Selected(i) = True
        Else
            Me.特徴選択リスト.Selected(i) = False
        End If
    Next i

    Call KEISAN
    On Error GoTo Err_Form_Current
    Dim intNO As Integer
    Dim strNO As String * 2

Public Sub KEISAN()
    Dim db As DAO.Database
    Set db = CurrentDb

    Dim KBN(6) As Integer
    Dim l As Integer
    Dim i As Integer
    Dim j As Integer
    Dim u As Integer
    Dim D As Integer

    If Not IsNumeric(Me.規格性量_仕上巾) Then Me.規格性量_仕上巾 = 0
    If Not IsNumeric(Me.規格性量_仕上長) Then Me.規格性量_仕上長 = 0
    If Not IsNumeric(Me.規格性量_仕上目付) Then Me.規格性量_仕上目付 = 0
    If Not IsNumeric(Me.規格性量_生機巾) Then Me.規格性量_生機巾 = 0
    If Not IsNumeric(Me.規格性量_生機長) Then Me.規格性量_生機長 = 0
    If Not IsNumeric(Me.規格性量_生機目付) Then Me.規格性量_生機目付 = 0
    If Not IsNumeric(Me.規格性量_生機目付平米) Then Me.規格性量_生機目付平米 = 0
    If Not IsNumeric(Me.規格性量_生機重量) Then Me.規格性量_生機重量 = 0
    If Not IsNumeric(Me.一反カウント) Then Me.一反カウント = 0
    If Not IsNumeric(Me.QR公称長) Then Me.QR公称長 = 0
    If Not IsNumeric(Me.QR公称巾) Then Me.QR公称巾 = 0
    If Not IsNumeric(Me.QR耳カット巾) Then Me.QR耳カット巾 = 0
    If Not IsNumeric(Me.QR反末ロス) Then Me.QR反末ロス = 0
    If Not IsNumeric(Me.QR精練原料率) Then Me.QR精練原料率 = 0
    If Not IsNumeric(Me.QRその他ロス率) Then Me.QRその他ロス率 = 0
    If Not IsNumeric(Me.通気量) Then Me.通気量 = 0
    If Not IsNumeric(Me.UPF) Then Me.UPF = 0
    If Not IsNumeric(Me.引裂強度タテ) Then Me.引裂強度タテ = 0
    If Not IsNumeric(Me.引裂強度ヨコ) Then Me.引裂強度ヨコ = 0
    If Not IsNumeric(Me.引張強度タテ) Then Me.引張強度タテ = 0
    If Not IsNumeric(Me.引張強度ヨコ) Then Me.引張強度ヨコ = 0
    If Not IsNumeric(Me.引張伸度タテ) Then Me.引張伸度タテ = 0
    If Not IsNumeric(Me.引張伸度ヨコ) Then Me.引張伸度ヨコ = 0

    For i = 0 To 6
        gcKeisan(i) = 0
        KBN(i) = 0
    Next i

''  生機目付平米
    If Me.規格性量_生機巾 = 0 Then
        Me.規格性量_生機目付平米 = 0
    Else
        Me.規格性量_生機目付平米 = Format(Me.規格性量_生機目付 / (Me.規格性量_生機巾 / 100), "0")
    End If

''  仕上長
    If (Me.設計区分 = 0) Or (Me.設計区分 = 3) Or (Me.設計区分 = 4) Or (Me.設計区分 = 5) Then
        Me.QR製品長 = Format(Me.QR公称長 * 1.01, "0.0")
        Me.規格性量_仕上長 = Me.QR製品長
    End If

''  仕上目付平米
    If Me.規格性量_仕上巾 = 0 Then
        Me.規格性量_仕上目付平米 = 0
    Else
        Me.規格性量_仕上目付平米 = Format(Me.規格性量_仕上目付 / (Me.規格性量_仕上巾 / 100), "0")
    End If

''  仕上目付け
    If Me.規格性量_仕上巾 = 0 Then Me.QR仕上目付 = 0 Else Me.QR仕上目付 = Format(Me.規格性量_仕上目付 / Me.規格性量_仕上巾 * 100, "0")

''  仕上重量
    Me.QR製品重量 = Fix((Me.規格性量_仕上目付 * Me.規格性量_仕上長 / 1000) * 100) / 100
    Me.規格性量_仕上重量 = Fix((Me.規格性量_仕上目付 * Me.規格性量_仕上長 / 1000) * 100) / 100
''  耳ロス
    If (Me.QR耳カット巾 + Me.規格性量_仕上巾) = 0 Then Me.QR耳カットロス率 = 0 Else Me.QR耳カットロス率 = Fix(((Me.QR耳カット巾 / (Me.QR耳カット巾 + Me.規格性量_仕上巾)) * 1000) + 0.5) / 10

''  反末ロス
    If Me.QR公称長 = 0 Then Me.QR反末ロス率 = 0 Else Me.QR反末ロス率 = Fix((Me.QR反末ロス / Me.QR公称長) * 1000) / 10

''  ロス率合計
    Me.QRロス率合計 = Format((1 - ((1 - (Me.QR精練原料率 / 100)) * (1 - (Me.QR反末ロス率 / 100)) * (1 - (Me.QRその他ロス率 / 100)) * (1 - (Me.QR耳カットロス率 / 100)))) * 100, "0.0")

''  生機重量
    If (Me.設計区分 = 1) Or (Me.設計区分 = 2) Then    '20110610 修正前　計算or手入力かの修正
        Me.QR生機重量 = Me.規格性量_生機重量
        Me.Caption = "設計書"          ''''20110610 追加　計算or手入力か？
    Else
        If Me.設計区分 = 0 Then
            If Me.QRロス率合計 = 100 Then
                Me.規格性量_生機重量 = Format(Fix(Me.QR製品重量), "0.00")
            Else
                Me.QR生機重量 = Format(Fix(Me.QR製品重量 / (1 - (Me.QRロス率合計 * 0.01)) * 100) / 100, "0.00")
                Me.規格性量_生機重量 = Format(Fix(Me.QR製品重量 / (1 - (Me.QRロス率合計 * 0.01)) * 100) / 100, "0.00")
            End If
            Me.Caption = "設計書　■計算式■"            ''''20110610 追加　計算or手入力か？
        Else
            Me.QR生機重量 = Me.規格性量_生機重量
            Dim tmp計算重量 As Currency
            If Me.QRロス率合計 = 100 Then
                tmp計算重量 = Format(Fix(Me.QR製品重量), "0.00")
            Else
                tmp計算重量 = Format(Fix(Me.QR製品重量 / (1 - (Me.QRロス率合計 * 0.01)) * 100) / 100, "0.00")
            End If
            If tmp計算重量 = Me.規格性量_生機重量 Then
                Me.Caption = "設計書　■計算式■"          ''''20110610 追加　計算or手入力か？
            Else
                Me.Caption = "設計書　◆手入力◆(計 " & Trim(Format(tmp計算重量, "0.00")) & " 手 " & Trim(Format(Me.規格性量_生機重量, "0.00")) & ")"          ''''20110610 追加　計算or手入力か？
            End If
        End If
    End If

    If Left(Me.生機品番 & Space(2), 2) = "ﾜJ" Then
        Me.出荷許容範囲_自 = Fix(Me.規格性量_生機重量 * 100 * 0.98) / 100
        Me.出荷許容範囲_至 = Fix(Me.規格性量_生機重量 * 100 * 1.02) / 100
    Else
        If Me.メーカーCD = 0 Then
            Me.出荷許容範囲_自 = Fix(Me.規格性量_生機重量 * 100 * 0.97) / 100
            Me.出荷許容範囲_至 = Fix(Me.規格性量_生機重量 * 100 * 1.03) / 100
        Else
            Me.出荷許容範囲_自 = Fix(Me.規格性量_生機重量 * 100 * 0.95) / 100
            Me.出荷許容範囲_至 = Fix(Me.規格性量_生機重量 * 100 * 1.05) / 100
        End If
    End If

''  ＱＲ生機長
    If Me.規格性量_生機目付 = 0 Then Me.QR生機長 = 0 Else Me.QR生機長 = Format(Me.QR生機重量 / Me.規格性量_生機目付 * 1000, "0.0")

''  ＱＲ正味糸量
    Dim WK正味MAX   As Double
    Dim WK正味MAX_I As Integer
    Me.QR総正味糸量 = 0

    On Error Resume Next
    DoCmd.RunCommand acCmdSaveRecord
    On Error GoTo 0

    Dim rs1 As DAO.Recordset
    Set rs1 = db.OpenRecordset("設計Ｂワーク")
    rs1.MoveFirst

    i = 0

    Do Until rs1.EOF
        i = i + 1
        If Not IsNull(rs1![原糸区分]) And rs1![原糸区分] <> "X" Then
            rs1.Edit
            If KBN(Asc(rs1![原糸区分]) - 64) = 0 Then
                KBN(Asc(rs1![原糸区分]) - 64) = rs1![設計明細NO]
            End If
            u = 0
            D = 0
            For j = 1 To 8
                u = u + IIf(rs1("給糸ＮＯＵ0" & j) > 0, 1, 0)
                D = D + IIf(rs1("給糸ＮＯＤ0" & j) > 0, 1, 0)
            Next j
            gcKeisan(KBN(Asc(rs1![原糸区分]) - 64)) = gcKeisan(KBN(Asc(rs1![原糸区分]) - 64)) + (rs1![見掛繊度] * rs1![給糸本数] * u * rs1![ループ長1]) + (rs1![見掛繊度] * rs1![給糸本数] * D * rs1![ループ長2])
            gcKeisan(0) = gcKeisan(0) + (rs1![見掛繊度] * rs1![給糸本数] * u * rs1![ループ長1]) + (rs1![見掛繊度] * rs1![給糸本数] * D * rs1![ループ長2])
            rs1![QR交編率] = rs1![交編率]
            rs1![QR正味糸量] = Format(Fix(Me.QR生機重量 * rs1![交編率]) / 100, "0.00")
            If WK正味MAX < rs1![QR正味糸量] Then
                WK正味MAX = rs1![QR正味糸量]
                WK正味MAX_I = 0
            End If
            Me.QR総正味糸量 = Me.QR総正味糸量 + rs1![QR正味糸量]
            rs1.Update
        End If
        rs1.MoveNext
    Loop

    Me.QR総正味糸量 = Format(Me.QR総正味糸量, "0.000")

    On Error Resume Next
    DoCmd.RunCommand acCmdSaveRecord
    On Error GoTo 0

    '総持掛糸量
    Me.QR総持掛糸量 = 0
    Dim rs2 As DAO.Recordset
    Set rs2 = db.OpenRecordset("設計Ｂワーク")
    rs2.MoveFirst
    i = 0
    Do Until rs2.EOF
        i = i + 1
        If Not IsNull(rs2![原糸区分]) And rs2![原糸区分] <> "X" Then
            rs2.Edit
            rs2![QR持掛糸量] = Format(rs2![QR正味糸量] * (1 + (rs2![QR編立ロス率] * 0.01)), "0.00")
            Me.QR総持掛糸量 = Me.QR総持掛糸量 + rs2![QR持掛糸量]
            rs2.Update
        End If
        rs2.MoveNext
    Loop

    Me.QR総持掛糸量 = Format(Me.QR総持掛糸量, "0.000")

    ''持掛目付
    If Me.QR公称長 = 0 Then
        Me.QR持掛目付 = 0
    Else
        Me.QR持掛目付 = Fix(Me.QR総持掛糸量 / Me.QR公称長 * 10000) / 10
    End If

    ''仕上Ｗ
    If Me.規格性量_仕上巾 = 0 Then
        Me.QR製品ウェル = 0
    Else
        Me.QR製品ウェル = Format(Me.編機針数 / (Me.規格性量_仕上巾 + 3) * 2.54, "0.0")
    End If

    Call 生産量計算
    DoCmd.RunCommand acCmdSaveRecord

    Me.設計Bメンテ.Requery
End Sub

Private Function 採番() As Long
    Dim db As DAO.Database
    Dim rs As DAO.Recordset

    Set db = CurrentDb
    Set rs = db.OpenRecordset("SELECT カウント FROM 設計ＣＴマスタ WHERE コード='00'", dbOpenDynaset)
    If Not rs.EOF Then
        rs.Edit
        採番 = rs.Fields("カウント")
        rs.Fields("カウント") = rs.Fields("カウント") + 1
        rs.Update
    Else
        採番 = 0
    End If
    rs.Close
    Set rs = Nothing

    db.Execute "UPDATE 設計Ｂワーク SET 設計ＩＤ = " & 採番 & _
               " WHERE 設計ＩＤ = " & Me![設計ID] & ";", dbFailOnError

    Set db = Nothing
End Function

Private Sub 生産量計算()
    Rem 生機長
    If Me.規格性量_生機目付 = 0 Then
        Me.規格性量_生機長 = 0
    Else
        Me.規格性量_生機長 = Fix((Me.規格性量_生機重量 / (Me.規格性量_生機目付 / 1000) + 0.05) * 10) / 10
    End If
    Rem 月産
    Dim WK_KYUSI_TBL(1 To 99) As Integer
    Dim WK_KYUSI_NO(1 To 6, 1 To 99) As Integer
    Dim WK_KYUSI_HON(1 To 6) As Integer
    Dim WK_MIKAKE_SENNDO(1 To 6) As Integer
    Dim WK_LOOP1(1 To 6) As Integer
    Dim WK_LOOP2(1 To 6) As Integer
    Dim WK_KYUSI_KUTI_ALL As Integer
    Dim WK生産量K月 As Double
    Dim l As Integer
    Dim i As Integer
    Erase WK_KYUSI_TBL
    Erase WK_KYUSI_NO
    Erase WK_KYUSI_HON
    Erase WK_MIKAKE_SENNDO
    Erase WK_LOOP1
    Erase WK_LOOP2
    WK生産量K月 = 0
    WK_KYUSI_KUTI_ALL = 0
    Dim rs As DAO.Recordset
    Dim f As Form
    Set f = Me.設計Bメンテ.Form
    Set rs = f.RecordsetClone
    rs.MoveFirst
    i = 0
    Do Until rs.EOF
        i = i + 1
        If VarType(rs![ループ長1]) = vbCurrency Then
            WK_LOOP1(i) = rs![ループ長1]
        End If
        If VarType(rs![ループ長2]) = vbCurrency Then
            WK_LOOP2(i) = rs![ループ長2]
        End If
        If VarType(rs![給糸本数]) = vbByte Then
            WK_KYUSI_HON(i) = rs![給糸本数]
        End If
        If VarType(rs![見掛繊度]) = vbCurrency Then
            WK_MIKAKE_SENNDO(i) = rs![見掛繊度]

Private Sub Form_KeyDown(KeyCode As Integer, Shift As Integer)
    If Shift = 0 Then
        Select Case KeyCode
        Case 112    'F1
            KeyCode = 0
            If Me.修正.Enabled Then
                Call 修正_Click
            End If
        Case 113    'F2
            KeyCode = 0
            If Me.リプロ.Enabled Then
                Call リプロ_Click
            End If
        Case 114    'F3
            KeyCode = 0
            If Me.複写.Enabled Then
                Call 複写_Click
            End If
        Case 115    'F4
            KeyCode = 0
            If Me.印刷.Enabled Then
                Call 印刷_Click
            End If
        Case 116    'F5
            KeyCode = 0
            If Me.閉じる.Enabled Then
                Call 閉じる_Click
            End If
        Case 27     'ESC
            KeyCode = 0
            If Me.Dirty Then
                On Error Resume Next
                DoCmd.RunCommand acCmdUndo
                On Error GoTo 0
                If Not Me.Dirty Then
                    Call Form_Current
                End If
            End If
        End Select
    End If
    If Shift = 1 Then
        Select Case KeyCode
        Case 112    '+F1
            KeyCode = 0
            If Me.削除.Enabled Then
                Call 削除_Click
            End If
        Case 113    '+F2
            KeyCode = 0
            If IsNull(組織図のコード表示) Or 組織図のコード表示 = 0 Then
                組織図のコード表示 = True
            Else
                組織図のコード表示 = 0
            End If
            If Me.組織図のコード表示.Enabled Then
                Call 組織図のコード表示_Click
            End If
        Case 114    '+F3
            KeyCode = 0
            If Me.組織図一覧.Enabled Then
                Call 組織図一覧_Click
            End If
        Case 115    '+F4
            KeyCode = 0
            If Me.トグル計測値糸量.Enabled Then
                Call トグル計測値糸量_Click
            End If
        Case 116    '+F5
            KeyCode = 0
            If Me.並び替え.Enabled Then
                Call 並び替え_Click
            End If
        End Select
    End If
End Sub
