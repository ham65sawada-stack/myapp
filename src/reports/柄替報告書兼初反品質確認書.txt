    G釜 = 編機ゲージ & "-" & Me.編機釜経
    針本数 = Me.編機針数
    口数 = Me.給糸口数
    非連結使用口数 = Me.給糸本数    ''設計Ｈの項目名「給糸本数」の内容が使用口数の為。
    規格値幅 = Me.規格性量_生機巾
    規格値長 = Me.規格性量_生機長
    規格値重量 = Me.規格性量_生機重量
    規格値目付 = Me.規格性量_生機目付
    回転数 = Me.RPM
    一反カウンター = Me.一反カウント
    日付年 = Format(Me.作成日付, "YY")
    日付月 = Format(Me.作成日付, "MM")
    日付日 = Format(Me.作成日付, "DD")
    編工場名 = LINK_ATESAKI
    If Left(LINK_ATESAKI, 4) = "ミツカワ" Or Trim(LINK_ATESAKI) = "" Then
        非連結号機 = LINK_GOUKI
    Else
        非連結号機 = ""
    End If
    ウエル = Me.ウエル数
    sSQL = "SELECT * FROM 設計Ｂ"
    sSQL = sSQL & " WHERE 設計ＩＤ=" & Me.設計ID
    sSQL = sSQL & " ORDER BY 原糸区分,原糸品番,給糸ＮＯＵ01"
    Set f_rstSekkeiB = f_dbsDB.OpenRecordset(sSQL, dbOpenSnapshot)
    With f_rstSekkeiB
        iIndex = 1
        Do While Not .EOF
            sIndex = Trim(str(iIndex))

            If .Fields("原糸区分") <> "X" Then
                Me.Controls("原糸品種ＣＤ" & sIndex) = .Fields("原糸区分")
                Me.Controls("原糸品種名" & sIndex) = Space(4) & .Fields("原糸コメント") & vbCrLf & .Fields("原糸品番")
                Me.Controls("給糸№" & sIndex) = Null
                For iKyusi = 1 To 8
                    sKyusi = Format(iKyusi, "00")
                    Me.Controls("給糸№" & sIndex) = Me.Controls("給糸№" & sIndex) & Right(Space(3) & Format(.Fields("給糸ＮＯＵ" & sKyusi), "##"), 3)
                Next iKyusi
                Me.Controls("給糸№" & sIndex) = Me.Controls("給糸№" & sIndex) & vbCrLf
                For iKyusi = 1 To 8
                    sKyusi = Format(iKyusi, "00")
                    Me.Controls("給糸№" & sIndex) = Me.Controls("給糸№" & sIndex) & Right(Space(3) & Format(.Fields("給糸ＮＯＤ" & sKyusi), "##"), 3)
                Next iKyusi
                Me.Controls("設計長" & sIndex) = Format(.Fields("ループ長１"), "###") & vbCrLf & Format(.Fields("ループ長２"), "###")
                Me.Controls("計測長" & sIndex) = Null
                Me.Controls("設計率" & sIndex) = Format(.Fields("交編率"), "##0.00;##0.00;#")
            Else
                Me.Controls("原糸品種ＣＤ" & sIndex) = Null
                Me.Controls("原糸品種名" & sIndex) = Null
                Me.Controls("給糸№" & sIndex) = Null
                Me.Controls("設計長" & sIndex) = Null
                Me.Controls("計測長" & sIndex) = Null
                Me.Controls("設計率" & sIndex) = Null
            End If

            iIndex = iIndex + 1

            .MoveNext
        Loop

        .Close
    End With

    Set f_rstSekkeiB = Nothing
End Sub
