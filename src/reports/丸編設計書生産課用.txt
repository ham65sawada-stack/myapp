Private Sub 詳細_Format(Cancel As Integer, FormatCount As Integer)

    Dim i As Integer

    i = Len(Trim(LINK_ATESAKI))
    If i < 14 Then
        Me.宛先名14 = LINK_ATESAKI
    ElseIf i < 17 Then
        Me.宛先名12 = LINK_ATESAKI
    Else
        Me.宛先名10 = LINK_ATESAKI
    End If

''''''''''    If Me.QR公称巾 <> 0 Or Me.QR公称長 <> 0 Then
''''''''''        Me![非連結公称規格] = Me.QR公称巾 & "×" & Me.QR公称長
''''''''''    Else
''''''''''        Me![非連結公称規格] = ""
''''''''''    End If

    If Trim(LINK_ATESAKI) = "" Then
        Me![号機] = LINK_GOUKI
''''''''''        Me.非連結編立工場名 = Me.編立工場名
        Me.r備考左 = Me.備考左
        Me.r備考右 = Me.備考右
    Else
        Me![号機] = ""
'''''''''        Me.非連結編立工場名 = ""

        Dim vl As Variant
        vl = Split(Me.備考左, vbCrLf)
        Dim ml As Long
        ml = UBound(vl)
        Dim bl As String
        bl = ""
        For i = 0 To ml
            If Left(vl(i), 4) <> "MM0." Then
                bl = bl & vl(i) & vbCrLf
            End If
        Next i
        Me.r備考左 = bl

        Dim vr As Variant
        vr = Split(Me.備考右, vbCrLf)
        Dim mr As Long
        mr = UBound(vr)
        Dim br As String
        br = ""
        For i = 0 To mr
            If Left(vr(i), 4) <> "MM0." Then
                br = br & vr(i) & vbCrLf
            End If
        Next i
        Me.r備考右 = br
    End If
    On Error GoTo Err_詳細_Format
    Dim intNO As Integer
    Dim strNO As String * 2
    For intNO = 1 To 24
        strNO = Format(intNO, "00")
        If LINK_設計書印刷時の組織図有無 Then
            Me("組織図" & strNO).PictureData = IIf(Me("組織CD" & strNO) = 0, イメージ空白.PictureData, DLookup("組織図DATA", "組織図マスタ", "組織CD=" & Me("組織CD" & strNO)))
        Else
            Me("組織図" & strNO).PictureData = イメージ空白.PictureData
        End If
    Next intNO

Exit_詳細_Format:
    Exit Sub

Err_詳細_Format:
    Resume Next
End Sub
