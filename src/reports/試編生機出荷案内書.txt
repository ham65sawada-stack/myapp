Option Compare Database
Option Explicit

Private f_dbsDB         As DAO.Database
Private f_rstSiami      As DAO.Recordset
Private f_rstSekkeiH    As DAO.Recordset
Private f_rstSekkeiB    As DAO.Recordset

Private Sub Report_Open(Cancel As Integer)
    Set f_dbsDB = CurrentDb
    Set f_rstSiami = Nothing
End Sub

Private Sub 詳細_Print(Cancel As Integer, PrintCount As Integer)
    Dim sIndex      As String
    Dim sFldIndex   As String
    Dim iIndex      As Integer
    Dim i      As Integer
    Dim sSQL        As String
    Dim s使用面     As String

    気付先名 = Me.写

    非連結出荷先名 = Me.出荷先名
    日付年 = Format(Me.伝票日付, "YY")
    日付月 = Format(Me.伝票日付, "MM")
    日付日 = Format(Me.伝票日付, "DD")
    非連結設計NO = Me.設計NO

    NO = 連番
    合計その他 = 0
    合計ハネ = 0
    合計重量 = 0
    合計針折 = 0
    合計汚れ = 0
    合計飛込 = 0
    合計長 = 0
    合計罰点計 = 0
    合計目付 = Null
    合計落傷 = 0

    sSQL = "SELECT * FROM 試編出荷データワーク"
    Set f_rstSiami = f_dbsDB.OpenRecordset(sSQL, dbOpenSnapshot)
    If f_rstSiami.EOF Then
        Exit Sub
    End If

    備考 = f_rstSiami.Fields("備考")

    For iIndex = 1 To 10
        sIndex = Format(iIndex, "00")
        sFldIndex = StrConv(Trim$(str(iIndex)), vbWide)

        With f_rstSiami
            If Not IsNull(.Fields("反番" & sFldIndex)) And Trim(.Fields("反番" & sFldIndex)) > "" Then
                Me.Controls("品番" & sIndex) = .Fields("品番" & sFldIndex)
                Me.Controls("反番" & sIndex) = .Fields("反番" & sFldIndex)
                Me.Controls("巾" & sIndex) = Format(.Fields("巾" & sFldIndex), "0")
                Me.Controls("長さ" & sIndex) = Format(.Fields("長" & sFldIndex), "0.0")
                Me.Controls("重量" & sIndex) = Format(.Fields("重量" & sFldIndex), "0.0")
                Me.Controls("目付" & sIndex) = Format(.Fields("目付" & sFldIndex), "0")
                Me.Controls("落傷" & sIndex) = Format(.Fields("落傷" & sFldIndex), "0;0;#")
                Me.Controls("針折" & sIndex) = Format(.Fields("針折" & sFldIndex), "0;0;#")
                Me.Controls("ハネ" & sIndex) = Format(.Fields("ハネ" & sFldIndex), "0;0;#")
                Me.Controls("汚れ" & sIndex) = Format(.Fields("汚れ" & sFldIndex), "0;0;#")
                Me.Controls("飛込" & sIndex) = Format(.Fields("飛込" & sFldIndex), "0;0;#")
                Me.Controls("その他" & sIndex) = Format(.Fields("その他" & sFldIndex), "0;0;#")
                Me.Controls("カット" & sIndex) = Format(.Fields("カット" & sFldIndex), "0;0;#")
                Me.Controls("罰点計" & sIndex) = Format(.Fields("落傷" & sFldIndex) + .Fields("針折" & sFldIndex) + .Fields("ハネ" & sFldIndex) + .Fields("その他" & sFldIndex), "0;0;0")
                Me.Controls("備考" & sIndex) = .Fields("明細備考" & sFldIndex)
                合計その他 = 合計その他 + val(IfNumeric(.Fields("その他" & sFldIndex)))
                合計ハネ = 合計ハネ + val(IfNumeric(.Fields("ハネ" & sFldIndex)))
                合計汚れ = 合計汚れ + val(IfNumeric(.Fields("汚れ" & sFldIndex)))
                合計飛込 = 合計飛込 + val(IfNumeric(.Fields("飛込" & sFldIndex)))
                合計重量 = 合計重量 + val(IfNumeric(.Fields("重量" & sFldIndex)))
                合計針折 = 合計針折 + val(IfNumeric(.Fields("針折" & sFldIndex)))
                合計長 = 合計長 + val(IfNumeric(.Fields("長" & sFldIndex)))
                合計罰点計 = 合計罰点計 + val(IfNumeric(.Fields("罰計" & sFldIndex)))
                合計落傷 = 合計落傷 + val(IfNumeric(.Fields("落傷" & sFldIndex)))
            Else
                Me.Controls("品番" & sIndex) = Null
                Me.Controls("反番" & sIndex) = Null
                Me.Controls("巾" & sIndex) = Null
                Me.Controls("長さ" & sIndex) = Null
                Me.Controls("重量" & sIndex) = Null
                Me.Controls("目付" & sIndex) = Null
                Me.Controls("落傷" & sIndex) = Null
                Me.Controls("針折" & sIndex) = Null
                Me.Controls("ハネ" & sIndex) = Null
                Me.Controls("汚れ" & sIndex) = Null
                Me.Controls("飛込" & sIndex) = Null
                Me.Controls("その他" & sIndex) = Null
                Me.Controls("カット" & sIndex) = Null
                Me.Controls("罰点計" & sIndex) = Null
                Me.Controls("備考" & sIndex) = Null
            End If
        End With

    Next iIndex
    合計重量 = Format(合計重量, "#,##0.0")
    合計長 = Format(合計長, "#,##0.0")

    '' 設計Ｈ
    sSQL = "SELECT 編機タイプ,編機ゲージ,編機釜経,使用面"
    sSQL = sSQL & " FROM 設計Ｈ"
    sSQL = sSQL & " WHERE 設計ID=" & Me.設計NO
    Set f_rstSekkeiH = f_dbsDB.OpenRecordset(sSQL, dbOpenSnapshot)

    With f_rstSekkeiH
        If Not .EOF Then
            機種 = .Fields("編機タイプ")
            ゲージ = .Fields("編機ゲージ")
            口径 = .Fields("編機釜経")
            s使用面 = Choose(.Fields("使用面") + 1, "中表", "", "裏面")
        Else
            s使用面 = ""
        End If
        .Close
    End With
    Set f_rstSekkeiH = Nothing

    Dim b As String
    b = Me("備考") & vbCrLf & s使用面

    f_rstSiami.Close
    Set f_rstSiami = Nothing

    '' 設計Ｂ
    sSQL = "SELECT 原糸品番,SUM(交編率) AS SUM交編率"
    sSQL = sSQL & " FROM 設計Ｂ"
    sSQL = sSQL & " WHERE 設計ID=" & Me.設計NO & " AND 原糸区分<>'X'"
    sSQL = sSQL & " GROUP BY 原糸区分,原糸品番"
    Set f_rstSekkeiB = f_dbsDB.OpenRecordset(sSQL, dbOpenSnapshot)
    With f_rstSekkeiB
        iIndex = 1
        Do While Not .EOF
            sIndex = Format(iIndex, "00")
            Me.Controls("原糸品番" & sIndex) = .Fields("原糸品番")
            Me.Controls("交編率" & sIndex) = Format(.Fields("SUM交編率"), "0.00;0.00;#")
            Me.Controls("買糸区分" & sIndex) = Me("原糸備考" & iIndex)
            iIndex = iIndex + 1
            .MoveNext
        Loop
        .Close
    End With
    Set f_rstSekkeiB = Nothing
End Sub
